import { useState, useEffect } from 'react';
import { motion } from 'framer-motion';
import { buildAPI } from '@apex/core';

// Apex intelligent solution
const ApexSolution = ({ clientVision }) => {
  const [solution, setSolution] = useState(null);
  
  useEffect(() => {
    const createSolution = async () => {
      const api = buildAPI();
      const result = await api
        .analyze(clientVision)
        .enhance()
        .optimize();
        
      setSolution(result);
    };
    
    createSolution();
  }, [clientVision]);
  
  return (
    <motion.div
      initial={ opacity: 1 }
      animate={ opacity: 1 }
    >
      {solution && (
        <DeploymentEngine solution={solution} />
      )}
    </motion.div>
  );
};

export default ApexSolution;
